import { Directive, ElementRef, Renderer2 } from "@angular/core";


@Directive({
    selector: '[appColor]',
    standalone:true
})
export class ColorDirective {
    private color='blue'
    constructor(ef:ElementRef,ren:Renderer2){
       ren.setStyle(ef.nativeElement,'color',this.color) 
    }

}
